\begin{frame}
  \frametitle{2. Normal Form}

  \begin{tanim}
    \alert{2NF}: anahtar olmayan her nitelik birincil anahtara baðýmlý
  \end{tanim}

  \begin{block}{1NF'den 2NF'ye geçiþ}
    \begin{itemize}
      \item 1NF'ye uyan $R$ baðýntýsýnda:
      \begin{itemize}
        \item $R(A, B, C, D)$, birincil anahtar: $\{A, B\}$
        \item $A \rightarrow D$
      \end{itemize}

      \pause
      \item 2NF olmasý için:
      \begin{itemize}
        \item $R1(A, D)$, birincil anahtar: $A$
        \item $R2(A, B, C)$, birincil anahtar: $\{A, B\}$\\
          $A$, $R1$'e baþvuran dýþ anahtar
      \end{itemize}
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{1NF-2NF Geçiþi Örneði}

  \begin{ornek}
    \begin{itemize}
      \item anahtar olmayan niteliklerden yalnýzca ORD birincil anahtara baðýmlý

      \pause
      \begin{itemize}
        \item $A$: MOVIEID
        \item $B$: ACTORID
        \item $C$: \{NAME,ORD\}
        \item $D$: \{TITLE, COUNTRY, LANGUAGE\}
      \end{itemize}
    \end{itemize}
  \end{ornek}
\end{frame}

\begin{frame}
  \frametitle{1NF-2NF Geçiþi Örneði}

  \begin{ornek}
    \begin{itemize}
      \item R1(MOVIEID, TITLE, COUNTRY, LANGUAGE)\\
        birincil anahtar: MOVIEID

      \pause
      \item R2(MOVIEID, ACTORID, NAME, ORD)\\
        birincil anahtar: \{MOVIEID, ACTORID\}\\
        MOVIEID, R1'e baþvuran dýþ anahtar
    \end{itemize}
 \end{ornek}
\end{frame}

\subsection{3. Normal Form}

\begin{frame}
  \frametitle{3. Normal Form}

  \begin{tanim}
    \alert{3NF}: anahtar olmayan nitelikler birincil anahtar dýþýnda bir nitelik
      ya da nitelik grubuna baðýmlý deðil
  \end{tanim}

  \pause
  \begin{block}{2NF'den 3NF'ye geçiþ}
    \begin{itemize}
      \item 2NF'ye uyan $R$ baðýntýsýnda:
      \begin{itemize}
        \item $R(A, B, C, D)$, birincil anahtar: $A$
        \item $C \rightarrow D$
      \end{itemize}

      \pause
      \item 3NF olmasý için:
      \begin{itemize}
        \item $R1(C, D)$, birincil anahtar: $C$
        \item $R2(A, B, C)$, birincil anahtar: $A$\\
          $C$, $R1$'e baþvuran dýþ anahtar
      \end{itemize}
    \end{itemize}
 \end{block}
\end{frame}
